import sys

import re

def decryptstr(s):
    cmd=''
    cmdlst=s.strip().split('&')
    pat=re.compile(r"Chr\([0-9]*\)")
    for item in cmdlst:
        if 'Chr' in item:
            beg,end=item.find('('),item.find(')')
            numstr=item[beg+1:end]
            cmd+=chr(int(numstr))
        else:
            cmd+=item.replace('"','')
    cmd=cmd.replace(' ','')
    cmd=cmd.replace('=','')
    cmd=cmd.replace(';','')
    cmd=cmd.replace('<','')
    print(cmd)

ss=input("Input String:")
while len(ss)!=0:

    ss=input("Input String:")
