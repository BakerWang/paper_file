# -*- coding:utf-8 -*-

import string

#读取密文
clipertxt=open("fuckme.txt").read().decode("utf-8")

#读取去除ascii字符以外的密文字符集
mwcharset=set(clipertxt)-set(string.printable)

#将密文字符集作为key，英文字母作为value拼成字典
dict=dict(zip(mwcharset,string.letters))

#将密文中加密的字符替换成英文字母
for key in dict:
    clipertxt=clipertxt.replace(key,dict[key])
